#include <iostream>

using namespace std;

int countPairings();

bool areFriends[10][10];
bool taken[10];
int n; //학생수
int m; //짝

int main()
{
	int testCase;
	cin >> testCase;
	int x, y;
	while (testCase--) {
		cin >> n >> m;
		for (int i = 0; i < m; i++) { 
			cin >> x >> y;
			areFriends[x][y] = true;
			areFriends[y][x] = true;
		}
		cout << countPairings() << endl;
		for (int i = 0; i < 10; i++) {
			for (int j = 0; j < 10; j++) {
				areFriends[i][j] = false;
			}
		}
	}
}

int countPairings() {
	int firstFree = -1;
	for (int i = 0; i < n; i++) {
		if (!taken[i]) {
			firstFree = i;
			break;
		}
	}

	if (firstFree == -1) return 1; 

	int count = 0;
	for (int i = firstFree + 1; i < n; i++) { //가장 작은 수 다음 부터 끝까지 
		if (!taken[i] && areFriends[firstFree][i]) { //짝이 아니고 친구라면
			taken[i] = true;
			taken[firstFree] = true;
			count += countPairings(); // 모두 짝을 지으면 1을 리턴해서 더하고 하고 아니면 카운트 그대로 리턴
			taken[firstFree] = false;
			taken[i] = false;
		}
	}
	return count;
}
