#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#define INF 9999
using namespace std;

bool isTwelve(const vector<int>& clocks);
int setClock(vector<int>& clocks, int swtch);
void pushSwitch(vector<int>& clocks, int swtch);
int min(int a, int b);

vector<vector<int>> v = { //스위치 
	{0, 1, 2},
	{3, 7, 9, 11},
	{4, 10, 14, 15},
	{0, 4, 5, 6, 7},
	{6, 7, 8, 10, 12},
	{0, 2, 14, 15},
	{3, 14 ,15},
	{4, 5, 7, 14, 15},
	{1, 2, 3, 4, 5},
	{3, 4, 5, 9, 13}
};

int main() {
	vector<int> clocks;
	int  testCase;
	cin >> testCase;
	while (testCase--) {
		for (int i = 0; i < 16; i++) {
			int tmp;
			cin >> tmp;
			clocks.push_back(tmp);
		}
		cout << setClock(clocks, 0);		
	}
	return 0;
}

int min(int a, int b) {
	return a < b ? a : b;
}

bool isTwelve(const vector<int>& clocks) {
	for (int i = 0; i < 16; i++) {
		if (clocks[i] != 12) return false;
	}
	return true;
}

void pushSwitch(vector<int>& clocks, int swtch) {
	for (int i = 0; i < v[swtch].size(); i++) {
		if (clocks[v[swtch][i]] == 12) {
			clocks[v[swtch][i]] = 3;
		}
		else {
			clocks[v[swtch][i]] += 3;
		}
	}
}

int setClock(vector<int>& clocks, int swtch) {
	if (swtch == 10) return isTwelve(clocks) ? 0 : INF;
	int ret = INF;
	for (int cnt = 0; cnt < 4; cnt++) {
		ret = min(ret, cnt + setClock(clocks, swtch + 1));
		pushSwitch(clocks, swtch);
	}
	return ret;
}
